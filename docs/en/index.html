Clone Repository

cd HaluCatch
2

Run Audit

#自动检测语言（中文/英文）\npython3 halucatch_core.py --skill-dir /path/to/your-skill
#或强制英文输出\npython3 halucatch_core.py --skill-dir /path/to/your-skill --lang en
#或使用包方式\npython3 -m halucatch --skill-dir /path/to/your-skill
